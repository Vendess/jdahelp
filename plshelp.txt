import requests

url = "link2" 

otazky = [
    "p = plt.hist([age1, age2]) do tohto histogramu chcem x os rozdelit po 5 a nie po 10 pomocou pomocou xticks a range"
]
for msg in otazky:
    try:
        r = requests.post(url, json={"message": msg}, timeout=10)
        r.raise_for_status()
        reply = r.json().get("reply", "")
        print(f"\nTy: {msg}\n ChatGPT: {reply}")
    except Exception as e:
        print(f"\n Chyba pri otázke: {msg}\n{e}")
